5. book_rating      Number          Рейтин

10.book_picture     String          Картинка
11.book_discount    int             Скинда
12.book_bestseller  bool            Бестселер да/нет
13.book_pages       int             Кол-во страниц
14.book_genres      Array           Жанры
15.book_age_limit   int             Возрастное ограничение
